Core Algorithms The Engine

🔢 1. Levenshtein Distance – Fuzzy Matching

Measures the minimum number of single‑character edits needed to change one string into another.

JavaScript 
function levenshteinDistance(a, b) {
    const aLen = a.length, bLen = b.length;
    if (aLen === 0) return bLen;
    if (bLen === 0) return aLen;

    const matrix = [];
    for (let i = 0; i <= bLen; i++) matrix[i] = [i];
    for (let j = 0; j <= aLen; j++) matrix[0][j] = j;

    for (let i = 1; i <= bLen; i++) {
        for (let j = 1; j <= aLen; j++) {
            const cost = a[j-1] === b[i-1] ? 0 : 1;
            matrix[i][j] = Math.min(
                matrix[i-1][j] + 1,      // deletion
                matrix[i][j-1] + 1,      // insertion
                matrix[i-1][j-1] + cost  // substitution
            );
        }
    }
    return matrix[bLen][aLen];
}
Example: "Comuter" → "Computer" → 1 edit (insert 'p') → 87.5% similarity

📧 3. Pattern Validation – Regex & Auto‑Correction

const RuleValidators = {
    email: (val) => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(val),
    phone: (val) => /^(\+?6?01)[0-9]{8,9}$/.test(val.replace(/\s/g, '')),
    icnumber: (val) => /^\d{12}$/.test(val),
    date: (val) => normalizeDateValue(val) !== null
};

// Auto‑correction
function fixEmail(value) {
    let corrected = value.toLowerCase().trim();
    if (!corrected.includes('@')) {
        corrected = corrected.replace(/\s+at\s+/, '@');
    }
    if (corrected.includes('@') && !corrected.includes('.')) {
        const [local, domain] = corrected.split('@');
        corrected = `${local}@${domain}.com`;
    }
    return RuleValidators.email(corrected) ? corrected : null;
}

🎯 4. Smart Imputation – Program → Faculty

When the faculty column is empty, the system automatically infers it from the program name.

JavaScript 
const PROGRAM_FACULTY_MAP = {
    'computer science': 'Faculty of Computing',
    'software engineering': 'Faculty of Computing',
    'business administration': 'Faculty of Business',
    'marketing': 'Faculty of Business'
};

function getProgramFacultySuggestion(row) {
    const progIdx = headers.findIndex(h => h.toLowerCase().includes('program'));
    const facIdx = headers.findIndex(h => h.toLowerCase().includes('faculty'));
    if (progIdx === -1 || facIdx === -1) return null;

    const program = String(row[progIdx]).toLowerCase();
    for (const [key, value] of Object.entries(PROGRAM_FACULTY_MAP)) {
        if (program.includes(key)) return { facIdx, suggestion: value };
    }
    return null;
}

AI Engine Intelligence

🧠 1. Learning from Corrections (localStorage)

The system remembers every accepted/rejected correction. Over time, it becomes smarter for each department.

JavaScript 
// Save correction
function saveCorrection(original, corrected, column) {
    const key = `${column}:${original.toLowerCase()}`;
    let learningData = JSON.parse(localStorage.getItem('ai_learning') || '{}');
    if (!learningData[key]) learningData[key] = {};
    learningData[key][corrected] = (learningData[key][corrected] || 0) + 1;
    localStorage.setItem('ai_learning', JSON.stringify(learningData));
}

// Retrieve learned correction (most frequent)
function getLearnedCorrection(value, column) {
    const data = JSON.parse(localStorage.getItem('ai_learning') || '{}');
    const corrections = data[`${column}:${value.toLowerCase()}`];
    if (!corrections) return null;
    const best = Object.entries(corrections).sort((a,b) => b[1] - a[1])[0];
    return { corrected: best[0], confidence: Math.min(95, 50 + best[1]*5) };
}

⚡ 3. Chunked Validation – 1000+ Rows Without Crashing

Processes rows in chunks of 50 using setTimeout to keep the browser responsive.

JavaScript 
function runValidation(showSummary = false, chunkSize = 50) {
    const totalRows = data.length;
    let start = 0;

    const processNextChunk = () => {
        const end = Math.min(start + chunkSize, totalRows);
        processRowChunk(start, end);
        start = end;
        updateUI();
        if (start < totalRows) {
            setTimeout(processNextChunk, 0);
        } else {
            recalculateStats();
            renderTable();
            renderErrorLog();
        }
    };
    processNextChunk();
}
